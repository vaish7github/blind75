class Solution{
    public static int result(int [] nums){
        int min= nums[0];
        int max= 0;
        for (int i =0;i<nums.length;i++){
            if (nums[i]<min){
                min= nums[i];
            }

            
            max= Math.max(max, nums[i]-min);
            
        }
        return max;
    }
}
